import { useCallback } from 'react'
import { useSession } from 'next-auth/react'
import { endpoints, config } from '@/lib/config'
import { getStatusMapping } from '@/lib/status-mapping'
import { ManuscriptDetailData, QualityCheck } from '@/types/manuscript-detail'
import { getImageUrl } from '@/lib/image-utils'
 
const buildApiUrl = (endpoint: string) => {
  const baseUrl = config.api.baseUrl.startsWith('http') ? '/api' : config.api.baseUrl
  return `${baseUrl}${endpoint}`
}
 
export function useManuscriptDetailApi() {
  const { data: session } = useSession()
 
  const fetchManuscriptDetail = useCallback(async (
    msid: string, 
    isFetching: boolean,
    setIsFetching: (value: boolean) => void,
    setIsLoadingApi: (value: boolean) => void
  ): Promise<ManuscriptDetailData | null> => {
    // Prevent multiple simultaneous calls
    if (isFetching) {
      return null
    }
    
    setIsFetching(true)
    setIsLoadingApi(true)
    
    if (!session) {
      console.error('❌ No session available for API call')
      setIsLoadingApi(false)
      setIsFetching(false)
      return null
    }
    
    try {
      const response = await fetch(`${buildApiUrl(endpoints.manuscriptDetails(msid))}?apiMode=true`, {
        headers: {
          'Content-Type': 'application/json',
          'Cookie': document.cookie,
        },
        credentials: 'include',
      })
 
      if (!response.ok) {
        const errorData = await response.json().catch(() => ({ error: 'Unknown error' }))
        throw new Error(`API request failed: ${response.status} - ${errorData.error || response.statusText}`)
      }
 
      const apiData: any = await response.json()
      
      // Check if this is an error response from API mode
      if (apiData.error && apiData.isApiMode) {
        throw new Error(`Data4Rev API Error: ${apiData.error} - ${apiData.details}`)
      }
      
      // Fetch validation data separately
      let validationData: any = null;
      try {
        const validationResponse = await fetch(`/api/v1/manuscripts/${msid}/validation`);
        if (validationResponse.ok) {
          validationData = await validationResponse.json();
        }
      } catch (validationError) {
        console.warn('Failed to fetch validation data:', validationError);
      }
 
      // Transform API data to match our manuscript format
      const statusMapping = getStatusMapping(apiData.status)
      
      // Helper function to add validation checks to figure
      const addValidationChecksToFigure = (figure: any, figureIndex: number, validationData: any) => {
        let figureChecks: QualityCheck[] = (figure.check_results || []).map((check: any) => ({
          id: `check-${figure.id}-${check.check_name || Math.random()}`,
          type: check.status === 'error' ? 'error' : check.status === 'warning' ? 'warning' : 'info',
          message: check.message || check.check_name || 'No message',
          details: check.details || check.message || check.check_name || 'No details available',
          aiGenerated: check.ai_generated !== undefined ? check.ai_generated : true,
          dismissed: false,
          figureTitle: figure.label || `Figure ${figure.id}`,
          panelId: check.panel_id || null
        }));
 
        // Add figure-level validation checks if available
        if (validationData && !validationData.fallback && validationData.figures) {
          const figureValidation = validationData.figures.find((f: any, index: number) => 
            f.label === figure.label || index === figureIndex
          );
          
          if (figureValidation) {
            // Add figure label checks
            if (figureValidation.label && Array.isArray(figureValidation.label)) {
              figureValidation.label.forEach((check: any) => {
                figureChecks.push({
                  id: `validation-${figure.id}-label-${Math.random()}`,
                  type: check.status === 'failed' ? 'error' : check.status === 'warning' ? 'warning' : 'info',
                  message: check.message || check.check_name || 'Figure label validation',
                  details: check.details || check.message || 'Figure label validation details',
                  aiGenerated: true,
                  dismissed: false,
                  figureTitle: figure.label || `Figure ${figure.id}`,
                  panelId: null
                });
              });
            }
          }
        }
 
        return figureChecks;
      }
 
      // Process figures
      const processedFigures = (apiData.figures || []).map((figure: any, figureIndex: number) => {
        const checks = addValidationChecksToFigure(figure, figureIndex, validationData);
        
        return {
          id: figure.id || `figure-${figureIndex}`,
          label: figure.label || `Figure ${figureIndex + 1}`,
          title: figure.title || figure.caption || figure.label || `Figure ${figureIndex + 1}`,
          legend: figure.legend || figure.caption || figure.description || '',
          panels: (figure.panels || []).map((panel: any, panelIndex: number) => ({
            id: panel.id || `${figure.id}-panel-${panelIndex}`,
            description: panel.description || panel.label || `Panel ${panelIndex + 1}`,
            legend: panel.legend || panel.caption || panel.description || '',
            imagePath: getImageUrl(panel.image_url || panel.imagePath, msid, {
              type: 'panel',
              panelId: panel.id
            }),
            coordinates: panel.coordinates,
            checks: panel.checks || []
          })),
          imagePath: getImageUrl(figure.image_url || figure.imagePath, msid, {
            type: 'full'
          }),
          checks
        }
      })
 
      const transformedData: ManuscriptDetailData = {
        msid: apiData.msid || msid,
        title: apiData.title || 'Unknown Title',
        authors: Array.isArray(apiData.authors) ? apiData.authors : (apiData.authors || '').split(',').map((a: string) => a.trim()),
        status: apiData.status || 'Unknown',
        displayStatus: statusMapping.displayStatus,
        priority: apiData.priority || 'normal',
        receivedDate: apiData.received_date || apiData.receivedDate || new Date().toISOString(),
        lastModified: apiData.last_modified || apiData.lastModified || new Date().toISOString(),
        figures: processedFigures,
        linkedData: apiData.linked_data || [],
        sourceData: apiData.source_data || [],
        notes: apiData.notes || '',
        dataAvailability: apiData.data_availability || '',
      }
 
      setIsLoadingApi(false)
      setIsFetching(false)
      return transformedData
 
    } catch (error) {
      console.error('❌ Failed to fetch manuscript details:', error)
      setIsLoadingApi(false)
      setIsFetching(false)
      throw error
    }
  }, [session])
 
  const depositManuscript = useCallback(async (msid: string) => {
    if (!session) {
      throw new Error('No session available for deposit')
    }
 
    const response = await fetch(`/api/v1/manuscripts/${msid}/deposit`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Cookie': document.cookie,
      },
      credentials: 'include',
    })
 
    if (!response.ok) {
      const errorData = await response.json().catch(() => ({ error: 'Unknown error' }))
      throw new Error(`Deposit failed: ${response.status} - ${errorData.error || response.statusText}`)
    }
 
    return await response.json()
  }, [session])
 
  const fetchFullText = useCallback(async (msid: string) => {
    if (!session) {
      throw new Error('No session available')
    }
 
    const response = await fetch(`/api/v1/manuscripts/${msid}/content`, {
      headers: {
        'Content-Type': 'application/json',
        'Cookie': document.cookie,
      },
      credentials: 'include',
    })
 
    if (!response.ok) {
      const errorData = await response.json().catch(() => ({ error: 'Unknown error' }))
      throw new Error(`Failed to fetch content: ${response.status} - ${errorData.error || response.statusText}`)
    }
 
    return await response.json()
  }, [session])
 
  return {
    fetchManuscriptDetail,
    depositManuscript,
    fetchFullText,
  }
}
